Easy S_ javascript Library

.append() Dom

Description:

append element to Dom

S_('.class').append(easy.create('h1', {'text': 'Hello World'}));

.aClass() Dom

Description:

Add Classes to Element

S_('.class').aClass('added-class');

.rmClass() Dom

Description:

Remove Classes from Element

S_('.class')rmClass('removed-class');

.css() Dom

Description:

modify style on selected element

S_('.class').css('background', '#222');
object as argument

you can define more than one style by pass object as argument

S_('.class').css({'background': '#222', 'border': '2px solid white'});

.html() Dom

Description:

grab the html from element

S_('.class').html();

.text() Dom

Description:

grab the text from element or set text

S_('.class').text(); S_('.class').text('added text');

.attr() Dom

Description:

set attribute on element

S_('.class').attr({'value': 'text'});

.rm() Dom

Description:

Removes element from dom

S_('.class').rm();

.on() Dom

Description:

Bind an event handler to javascript events like "click"

S_('.btn').on('click', handler);

.on() Dom

Description:

remove event handler from javascript events like "click"

S_('.btn').off('click', handler);

.filterJSON() Json

Description:

filter keys in Json Object

easy.filterJSON(json, 'Title', 'Genre', 'Runtime');

easy.Xcall() ajax

Description:

send Get and Post Request

easy.Xcall('GET', 'http://www.someurl.com/', callback);

.keyUp() dom

Description:

Bind an event handler to the "keyup" JavaScript event, or trigger that event on an element.

S_('input').keyUp(func, 13);
